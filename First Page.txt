    <script>
        function showUserMessages(userName) {
            var selectedUser = document.getElementById('selected-user');
            selectedUser.textContent = 'ข้อความของ ' + userName;

            var messagesContainer = document.getElementById('messages');
            messagesContainer.innerHTML = '';

            // สร้างข้อความสำหรับแต่ละคน
            var messages = [
                { sender: 'user1', text: 'ข้อความ 1 จาก user1' },
                { sender: 'user1', text: 'ข้อความ 2 จาก user1' },
                { sender: 'user2', text: 'ข้อความ 1 จาก user2' },
                { sender: 'user2', text: '<video src="user2_video.mp4" controls></video>' },
                { sender: 'user3', text: 'ข้อความ 1 จาก user3' },
                { sender: 'user3', text: 'ข้อความ 2 จาก user3' }
            ];

            messages.forEach(function(message) {
                if (message.sender === userName) {
                    var div = document.createElement('div');
                    div.classList.add('message');

                    var senderSpan = document.createElement('span');
                    senderSpan.classList.add('sender');
                    senderSpan.textContent = message.sender + ':';

                    var messageDiv = document.createElement('div');
                    messageDiv.innerHTML = message.text;

                    div.appendChild(senderSpan);
                    div.appendChild(messageDiv);

                    messagesContainer.appendChild(div);
                }
            });
        }
    </script>
